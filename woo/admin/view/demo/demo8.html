* @Ps(true,except={"a","b"}) # 当前控制器指定方法不需要验证权限

* @Forbid(only={"a","b"}) # 只有指定方法允许访问，其他不行
* @Forbid(except={"a","b"}) #指定方法禁止访问，其他的可以
* @Forbid(nodebug=true) # 仅开发模式下可以访问，部署模式不可以
*/
class Demo
{

    * @Except(method={"get"}) # 当前方法的get请求不登录也可以访问 其他请求方式需要（也可以通过配置文件实现）

    * @Ps(false) # 当前方法不需要验证权限
    * @Ps(as="delete") # 不需要单独设置权限 当前方法的权限和指定方法权限一致
    * @Ps(true,name="测试",as="delete")
    * @Forbid(only={"get"}) # 指定请求方式下才可以访问
    * @Forbid(except={"get"}) #除了指定方式不可以访问，其他都可以
    * @Forbid(only={"ajax","post"}) # 只有ajax且是post请求才可以访问

    */
